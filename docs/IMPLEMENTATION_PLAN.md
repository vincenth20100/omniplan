## Identified Risks

| ID | Risk | Impact | Mitigation |
|---|---|---|---|
| RISK-01 | Subproject cross-project links — `subprojectIds` field in `use-project.ts` opens additional Firestore subscriptions; mapping this to PostgreSQL requires recursive project queries | High | Defer subproject support; flatten to single-project for initial rebuild |
| RISK-02 | `converter/app.py` JVM startup at module load — in supervisord mode, if JVM fails the Flask process crashes silently | Medium | Add `/health` endpoint to Flask; supervisord restarts on failure; log to stdout |
| RISK-03 | PocketBase JWT validation in Route Handlers requires a live PocketBase instance even for local dev without auth | Medium | Add `AUTH_BYPASS=true` dev env var that skips token validation and uses a hardcoded dev user |
| RISK-04 | `project-selection-page.tsx` has 500+ lines of inline Firestore logic (project CRUD, invitation management) — requires full rewrite not just rewiring | High | T5 must treat this as a new file, not a patch |
| RISK-05 | Genkit Server Actions currently use `'use server'` directive — must be converted to standard Route Handlers to avoid Server Action limitations with streaming responses | Low | T8 converts them to Route Handlers; Server Actions kept only where appropriate |
| RISK-06 | SSE: one PostgreSQL LISTEN connection per connected user — at 100+ concurrent users this exhausts `max_connections` | Medium | Set `max_connections=200` in postgres config; use a connection pool (pgBouncer) if needed post-launch |
| RISK-07 | TypeScript error backlog — `ignoreBuildErrors: true` is currently masking an unknown number of type errors | High | T1 enables strict build; T3 and T5 must fix all errors in touched files; accept that T9 may require a cleanup pass |
| RISK-08 | `src/lib/omniplan-utils.ts` uses `NEXT_PUBLIC_OMNIPLAN_API_URL` in client-side code paths — the import flow must be fully server-side to remove this | Medium | T7 moves all converter calls to the `/api/import` Route Handler; client only calls `/api/import` |
| RISK-09 | Drizzle's `vector` column type requires `drizzle-orm` + `pg` to have `pgvector` extension enabled at migration time — must run `CREATE EXTENSION vector` before migrations | Low | T2 creates `0000_extensions.sql` as the first migration |

```sql
CREATE OR REPLACE FUNCTION notify_project_change()
RETURNS trigger AS $$
BEGIN
  PERFORM pg_notify(
    'omniplan_project_' || COALESCE(NEW.project_id, OLD.project_id)::text,
```

```typescript
import PocketBase from 'pocketbase';

// Client-side singleton
let _pb: PocketBase | null = null;

export function getPocketBase(): PocketBase {
  if (typeof window === 'undefined') {
    // Server-side: create a new instance per request (no singleton)
    return new PocketBase(process.env.POCKETBASE_URL ?? 'http://localhost:8090');
  }
  if (!_pb) {
    _pb = new PocketBase(
      process.env.NEXT_PUBLIC_POCKETBASE_URL ?? 'http://localhost:8090'
    );
    _pb.autoCancellation(false);
  }
  return _pb;
}
```

#### 3. Create `src/providers/auth-provider.tsx`

```typescript
'use client';

import React, { createContext, useContext, useEffect, useState } from 'react';
import { getPocketBase } from '@/lib/pocketbase';
import type { AppUser } from '@/types/auth';

interface AuthContextValue {
  user: AppUser | null;
  isLoading: boolean;
  login: (email: string, password: string) => Promise<void>;
  logout: () => Promise<void>;
}

const AuthContext = createContext<AuthContextValue | null>(null);

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const [user, setUser] = useState<AppUser | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    const pb = getPocketBase();

    // Restore session from localStorage (PocketBase handles this automatically)
    if (pb.authStore.isValid) {
      const record = pb.authStore.record;
      setUser({
        id: record?.id ?? '',
        email: record?.email ?? '',
        name: record?.name ?? record?.email ?? '',
        avatarUrl: record?.avatar ?? undefined,
        token: pb.authStore.token,
      });
    }
    setIsLoading(false);

    // Listen for auth state changes
    const unsubscribe = pb.authStore.onChange((token, record) => {
      if (token && record) {
        setUser({
          id: record.id,
          email: record.email ?? '',
          name: record.name ?? record.email ?? '',
          avatarUrl: record.avatar ?? undefined,
          token,
        });
      } else {
        setUser(null);
      }
    });

    return () => unsubscribe();
  }, []);

  const login = async (email: string, password: string) => {
    const pb = getPocketBase();
    await pb.collection('users').authWithPassword(email, password);
  };

  const logout = async () => {
    getPocketBase().authStore.clear();
  };

  return (
    <AuthContext.Provider value={{ user, isLoading, login, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth(): AuthContextValue {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth must be used within AuthProvider');
  return ctx;
}
```

```typescript
import PocketBase from 'pocketbase';
import { NextRequest } from 'next/server';
import type { AppUser } from '@/types/auth';

export async function requireAuth(req: NextRequest): Promise<AppUser> {
  // Dev bypass for local development without PocketBase running
  if (process.env.AUTH_BYPASS === 'true' && process.env.NODE_ENV !== 'production') {
    return { id: 'dev-user', email: 'dev@local', name: 'Dev User', token: 'dev' };
  }

  const authHeader = req.headers.get('authorization');
  if (!authHeader?.startsWith('Bearer ')) {
    throw new AuthError('Missing authorization header', 401);
  }

  const token = authHeader.slice(7);
  const pb = new PocketBase(process.env.POCKETBASE_URL ?? 'http://localhost:8090');
  pb.authStore.save(token, null);

  try {
    const record = await pb.collection('users').authRefresh();
    return {
      id: record.record.id,
      email: record.record.email ?? '',
      name: record.record.name ?? '',
      token: pb.authStore.token,
    };
  } catch {
    throw new AuthError('Invalid or expired token', 401);
  }
}

export class AuthError extends Error {
  constructor(message: string, public status: number) {
    super(message);
  }
}

export function authErrorResponse(err: unknown): Response {
  if (err instanceof AuthError) {
    return Response.json({ error: err.message }, { status: err.status });
  }
  return Response.json({ error: 'Internal server error' }, { status: 500 });
}
```

```typescript
export async function requireProjectAccess(
  userId: string,
  projectId: string,
  minRole: 'viewer' | 'editor' | 'owner' = 'viewer'
): Promise<void> {
  const [member] = await db
    .select()
    .from(projectMembers)
    .where(and(
      eq(projectMembers.projectId, projectId),
      eq(projectMembers.userId, userId)
    ));
  if (!member) throw new AuthError('Project not found or access denied', 404);
  const roleRank = { viewer: 0, editor: 1, owner: 2 };
  if (roleRank[member.role] < roleRank[minRole]) {
    throw new AuthError('Insufficient permissions', 403);
  }
}
```

```typescript
import { db } from '@/db';
import { tasks, links } from '@/db/schema';
import { eq } from 'drizzle-orm';
import { requireAuth, requireProjectAccess, authErrorResponse } from '@/lib/auth-middleware';

  req: NextRequest,
  { params }: { params: { projectId: string } }

  try {
    const user = await requireAuth(req);
    await requireProjectAccess(user.id, params.projectId, 'editor');

    const [allTasks, allLinks] = await Promise.all([
      db.select().from(tasks).where(eq(tasks.projectId, params.projectId)),
      db.select().from(links).where(eq(links.projectId, params.projectId)),
    ]);

    // calculateSchedule expects the internal Task[] and Link[] shape from lib/types.ts
    // Map DB rows → internal types, run CPM, map back → DB update
    const updatedTasks = calculateSchedule(allTasks as any, allLinks as any);

    // Batch update critical path fields
    await Promise.all(
      updatedTasks.map(t =>
        db.update(tasks)
          .set({
            startDate: t.startDate ? new Date(t.startDate) : null,
            finishDate: t.finishDate ? new Date(t.finishDate) : null,
            totalFloat: t.totalFloat ?? null,
            isOnCriticalPath: t.isCritical ?? false,
          })
          .where(eq(tasks.id, t.id))
      )
    );

    return Response.json({ updated: updatedTasks.length });
  } catch (err) {
    return authErrorResponse(err);
  }
}
```

```typescript
import { NextRequest } from 'next/server';
import postgres from 'postgres';
import { requireAuth, requireProjectAccess, authErrorResponse } from '@/lib/auth-middleware';

export const dynamic = 'force-dynamic';
export const runtime = 'nodejs'; // SSE requires Node.js runtime, not Edge

export async function GET(
  req: NextRequest,
  { params }: { params: { projectId: string } }
) {
  // SSE cannot set Authorization headers — accept token via query param
  const token = req.nextUrl.searchParams.get('token') ?? '';
  const fakeReq = new Request(req.url, {
    headers: { authorization: `Bearer ${token}` },
  });

  let user;
  try {
    user = await requireAuth(fakeReq as any);
    await requireProjectAccess(user.id, params.projectId);
  } catch (err) {
    return authErrorResponse(err);
  }

  const channel = `omniplan_project_${params.projectId}`;
  const sql = postgres(process.env.DATABASE_URL!);

  const stream = new ReadableStream({
    async start(controller) {
      const encoder = new TextEncoder();

      const send = (data: object) => {
        controller.enqueue(encoder.encode(`data: ${JSON.stringify(data)}\n\n`));
      };

      // Heartbeat every 25s to keep the connection alive through Cloudflare
      const heartbeat = setInterval(() => {
        controller.enqueue(encoder.encode(': heartbeat\n\n'));
      }, 25000);

      try {
        await sql.listen(channel, (payload) => {
          send(JSON.parse(payload));
        });
      } catch (err) {
        clearInterval(heartbeat);
        controller.close();
        await sql.end();
      }

      // Handle client disconnect
      req.signal.addEventListener('abort', async () => {
        clearInterval(heartbeat);
        await sql.end();
        controller.close();
      });
    },
  });

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive',
      'X-Accel-Buffering': 'no', // Disable Nginx buffering if present
    },
  });
}
```

```typescript
import { NextRequest } from 'next/server';
import { db } from '@/db';
import { projects, tasks, links, resources, assignments, projectMembers } from '@/db/schema';
import { requireAuth, authErrorResponse } from '@/lib/auth-middleware';
import { mapAnalyzeResponse } from '@/lib/omniplan-utils';

const CONVERTER_URL = process.env.CONVERTER_URL ?? 'http://localhost:7860';

export async function POST(req: NextRequest) {
  try {
    const user = await requireAuth(req);

    const formData = await req.formData();
    const file = formData.get('file') as File | null;
    if (!file) {
      return Response.json({ error: 'No file provided' }, { status: 400 });
    }

    // Forward to Flask converter
    const converterForm = new FormData();
    converterForm.append('file', file);
    const converterRes = await fetch(`${CONVERTER_URL}/analyze`, {
      method: 'POST',
      body: converterForm,
    });

    if (!converterRes.ok) {
      const err = await converterRes.text();
      return Response.json({ error: `Converter error: ${err}` }, { status: 502 });
    }

    const analyzeResult = await converterRes.json();

    // Map converter output to internal types using existing utility
    const { project: projectData, tasks: taskData, links: linkData, resources: resourceData } =
      mapAnalyzeResponse(analyzeResult);

    // Persist to PostgreSQL
    const [project] = await db.insert(projects).values({
      ownerId: user.id,
      name: projectData.name ?? file.name,
      settings: projectData.settings ?? {},
    }).returning();

    await db.insert(projectMembers).values({ projectId: project.id, userId: user.id, role: 'owner' });

    if (taskData.length > 0) {
      await db.insert(tasks).values(taskData.map(t => ({ ...t, projectId: project.id })));
    }
    if (linkData.length > 0) {
      await db.insert(links).values(linkData.map(l => ({ ...l, projectId: project.id })));
    }
    if (resourceData.length > 0) {
      await db.insert(resources).values(resourceData.map(r => ({ ...r, projectId: project.id })));
    }

    return Response.json({ projectId: project.id }, { status: 201 });
  } catch (err) {
    return authErrorResponse(err);
  }
}
```
